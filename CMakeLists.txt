cmake_minimum_required(VERSION 3.8)
project(ros2_pong_semiautomatic)

# Alap√©rtelmezett C++ verzi√≥

  rclcpp
  std_msgs
  geometry_msgs
)

# Fejl√©c mapp√°k

# üîß A saj√°t node-od: game_manager

# Telep√≠t√©s

  game_manager
  DESTINATION lib/${PROJECT_NAME}
)

# Launch f√°jlok (ha vannak)

  DESTINATION share/${PROJECT_NAME}
